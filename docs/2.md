    return "RoboCup!"

    return "Welcome to " + target()

## Object Oriented Python

```python
class BaseClass():
    globalVar = 1

    # Constructor
    def __init__(self):
	print("Initializing a Base")
	self.localVar = 2

class ChildClass(BaseClass):
    def __init__(self):
	super().__init__() # call superclass constructor
	print("Initializing Child")
	print("Our local var is: " + str(self.localVar))

a = BaseClass()
print("---")
a = ChildClass()
```

    Initializing a Base
    ---
    Initializing a Base
    Initializing Child
    Our local var is: 2
